// 1 float to fixed 3 RGB rainbow Function
fixed3 rainbow(float h) {
        h = fmod(fmod(h, 1.0) + 1.0, 1.0);
        float h6 = h * 6.0;
        float r = clamp(h6 - 4.0, 0.0, 1.0) +
            clamp(2.0 - h6, 0.0, 1.0);
        float g = h6 < 2.0
            ? clamp(h6, 0.0, 1.0)
            : clamp(4.0 - h6, 0.0, 1.0);
        float b = h6 < 4.0
            ? clamp(h6 - 2.0, 0.0, 1.0)
            : clamp(6.0 - h6, 0.0, 1.0);
        return fixed3(r, g, b);
    }
